#1700 Frost

leader_event = {
	id = dionysian_archon.1700
	title = dionysian_archon.1700.name
	desc = dionysian_archon.1700.desc
	picture = GFX_evt_frozen
	show_sound = event_default
	is_triggered_only = yes
	fire_only_once = yes

	event_window_type = leader_story
	event_picture_background = "GFX_leader_recruitment_bg_renowned"

	picture_event_data = {
		portrait = root
	}

	trigger = {
		owner = {
			any_planet_within_border = {
				is_planet_class = pc_frozen
				OR = {
					has_deposit = no
					has_deposit_for = shipclass_research_station
				}
			}
		}
	}

	immediate = {
		owner = { set_country_flag = dionysian_triggered_archon_frost_event } 
		owner = { 
			random_planet_within_border = {
				limit = {
					is_planet_class = pc_frozen
					OR = {
						has_deposit = no
						has_deposit_for = shipclass_research_station
					}
				}
				save_event_target_as = dionysian_archon_event_target_planet
			} 
		}
	}

	option = {
		name = dionysian_archon.1700.a
		add_experience = 2000
		event_target:dionysian_archon_event_target_planet = { add_deposit = dionysian_ice_realm_unity_deposit }
		owner = { add_modifier = { modifier = dionysian_ice_realm days = 3600 } }
	}

	option = {
		name = dionysian_archon.1700.b
		allow = {
			owner = { resource_stockpile_compare = { resource = energy value > 2000 } }
		}
		owner = { add_resource = { energy = -2000 } }
		event_target:dionysian_archon_event_target_planet = { change_pc = pc_arctic }
		hidden_effect = {
			event_target:dionysian_archon_event_target_planet = { reroll_planet = yes }
		}
	}

	option = {
		name = dionysian_archon.1700.c
		dionysian_refuse_archon_event = yes
	}
}

#2400 Hatred

leader_event = {
	id = dionysian_archon.2400
	title = dionysian_archon.2400.name
	desc = dionysian_archon.2400.desc
	picture = GFX_evt_alien_propaganda
	show_sound = event_default
	is_triggered_only = yes
	fire_only_once = yes

	event_window_type = leader_story
	event_picture_background = "GFX_leader_recruitment_bg_renowned"

	picture_event_data = {
		portrait = root
	}

	trigger = {
		owner = {
			any_owned_planet = {
				pop_amount > 500
			}
		}
	}

	immediate = { 
		owner = { 
			set_country_flag = dionysian_triggered_archon_hatred_event 

			random_owned_planet = {
				limit = {
					is_colony = yes
					pop_amount > 500
					exists = leader
					leader = { is_same_value = event_target:dionysian_choosing_planet_archon }
				}
				save_event_target_as = dionysian_archon_event_target_planet
			}

			if = {
				limit = {
					NOT = { exists = event_target:dionysian_archon_event_target_planet } 
			 	}
			 	random_owned_planet = {
					limit = {
						is_capital = no
						pop_amount > 500
					}
					save_event_target_as = dionysian_archon_event_target_planet
				}
			}

			if = {
				limit = {
					NOT = { exists = event_target:dionysian_archon_event_target_planet } 
			 	}
			 	capital_scope = { save_event_target_as = dionysian_archon_event_target_planet }
			}
		}
	}

	option = {
		name = dionysian_archon.2400.a
		TRIGGER = {	owner = { has_ethic = ethic_fanatic_militarist } }
		ALLOW = { event_target:dionysian_archon_event_target_planet = { pop_amount > 800 } }
		custom_tooltip = dionysian_archon.2400.a_tooltips
		add_experience = 2500
		owner = { 
			add_resource = { 
				sr_zro = 1500
			}
			add_modifier = { modifier = dionysian_imperial_martyrs days = 7200 } 
		}
		if = {
			limit = { owner = { NOT = { has_technology = tech_existential_campaigns } } }
			owner = {
				add_research_option = tech_existential_campaigns
				add_tech_progress = {
					tech = tech_existential_campaigns
					progress = 0.2
				}
			}
		}
		hidden_effect = {
			event_target:dionysian_archon_event_target_planet = {
				while = {
					count = 800
					random_owned_pop_group = {
						sacrifice_pop_group = { pop_group = this amount = 1 }
					}
				}
			}
		}
	}

	option = {
		name = dionysian_archon.2400.b
		ALLOW = { event_target:dionysian_archon_event_target_planet = { pop_amount > 500 } }
		custom_tooltip = dionysian_archon.2400.b_tooltips
		add_experience = 2000
		owner = { 
			add_resource = { 
				sr_zro = 1000
			}
			add_modifier = { modifier = dionysian_imperial_martyrs days = 3600 } 
		}
		if = {
			limit = { owner = { NOT = { has_technology = tech_existential_campaigns } } }
			owner = { 
				add_research_option = tech_existential_campaigns
				add_tech_progress = {
					tech = tech_existential_campaigns
					progress = 0.1
				}
			}
		}
		hidden_effect = {
			event_target:dionysian_archon_event_target_planet = {
				while = {
					count = 500
					random_owned_pop_group = {
						sacrifice_pop_group = { pop_group = this amount = 1 }
					}
				}
			}
		}
	}

	option = {
		name = dionysian_archon.2400.c
		dionysian_refuse_archon_event = yes
	}
}
